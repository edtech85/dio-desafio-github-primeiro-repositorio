	https://git-scm.com/download/win

***********************************************************************

-----------------------------------------------------------------------
	CLI - Command Line Interface
-----------------------------------------------------------------------
	
	-- Primeiramente chamar o repositório --
	$ cd /c/user/your_repository
	
	-- Em seguida --
	$ git init

-----------------------------------------------------------------------

	-- verificar status
	$ git status

-----------------------------------------------------------------------

	-- 1. para adicionar arquivos --
	$ git add .

	-- 2. commitar arquivo adicionado --

	$ git commit - m "escrever mensagem de comentário"

	-- 3. mandar para nuvem --
	$ git push origin main

-----------------------------------------------------------------------
	
	-- Para clonar repositório --
	$ git clone https://github.com/libgit2/libgit2

------------------------------------------------------------------

	-- Remover arquivo --
	$ rm PROJECTS.md
